DROP DATABASE IF EXISTS DB_ESCRITO;
CREATE DATABASE DB_ESCRITO;
USE DB_ESCRITO;

CREATE TABLE Persona(
CI INT UNSIGNED,
Nombre VARCHAR(40) NOT NULL,
Apellido VARCHAR(40) NOT NULL,
Telefono INT UNSIGNED NOT NULL,
PRIMARY KEY (CI)
);

CREATE TABLE Marca(
CIPersona INT UNSIGNED, 
FOREIGN KEY (CIPersona) REFERENCES Persona (CI),
FechaHoraEntrada VARCHAR(40),
FechaHoraSalida VARCHAR(40),
PRIMARY KEY (CIPersona)
);
